import { Plant } from '../types'; import { formatCurrency, addToCart } from '../utils'; /** * Renders an individual plant card component. * This component displays the plant image, details (price, sunlight, water), * and provides an "Add to Cart" button with visual feedback. * * @param plant - The plant object to display * @returns HTMLElement representing the plant card */ export function renderPlantCard(plant: Plant): HTMLElement { const article = document.createElement('article'); article.className = 'card group flex flex-col h-full animate-fade-in'; article.setAttribute('data-plant-id', plant.id); // Helper to get sunlight icon const getSunlightIcon = (level: string) => { switch (level) { case 'Direct': return '☀️'; case 'Bright Indirect': return '🌤️'; case 'Medium': return '⛅'; case 'Low': return '☁️'; default: return '☀️'; } }; // Helper to get water icon const getWaterIcon = (level: string) => { switch (level) { case 'High': return '💧💧💧'; case 'Medium': return '💧💧'; case 'Low': return '💧'; default: return '💧'; } }; article.innerHTML = `
${plant.isFeatured ? `
Featured
` : ''}

${plant.name}

${formatCurrency(plant.price)}

${plant.description}

${getSunlightIcon(plant.sunlight)} ${plant.sunlight}
${getWaterIcon(plant.water)} ${plant.water}
`; // Event Listener for Add to Cart const cartBtn = article.querySelector('.add-to-cart-btn') as HTMLButtonElement; const btnLabel = article.querySelector('.btn-label') as HTMLElement; if (cartBtn && btnLabel) { cartBtn.addEventListener('click', (e) => { e.preventDefault(); // Update global state addToCart(plant); // Visual Feedback const originalContent = cartBtn.innerHTML; const originalBg = cartBtn.style.backgroundColor; cartBtn.disabled = true; cartBtn.innerHTML = ` Added! `; cartBtn.classList.remove('btn-primary'); cartBtn.classList.add('bg-green-600', 'text-white'); // Revert after delay setTimeout(() => { cartBtn.disabled = false; cartBtn.innerHTML = originalContent; cartBtn.classList.add('btn-primary'); cartBtn.classList.remove('bg-green-600', 'text-white'); cartBtn.style.backgroundColor = originalBg; }, 1500); }); } return article; }
