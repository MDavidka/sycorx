import { generatePlantCatalog } from '../utils'; import { renderPlantCard } from './plant-card'; /** * Renders the Shop page view into the provided container. * This view generates a catalog of randomized plants and displays them in a responsive grid. * * @param container - The DOM element where the shop view will be rendered. */ export function renderShopView(container: HTMLElement): void { // 1. Clear container and set up the base layout container.innerHTML = `

Our Plant Collection

Explore our diverse range of lush greenery. From low-maintenance succulents to exotic tropicals, find the perfect companion for your living space.

Showing: 12 plants

Join the Green Thumb Club

Get 10% off your first order and expert plant care tips delivered to your inbox.

`; // 2. Generate the catalog and inject cards const grid = container.querySelector('#shop-grid'); const countDisplay = container.querySelector('#plant-count'); if (grid) { // Generate 12 random plants for the shop const catalog = generatePlantCatalog(12); if (countDisplay) { countDisplay.textContent = `${catalog.length} plants`; } // Render each plant card and append to the grid catalog.forEach(plant => { const card = renderPlantCard(plant); grid.appendChild(card); }); } }
