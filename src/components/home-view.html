import { Plant } from '../types'; import { generatePlantCatalog } from '../utils'; import { renderPlantCard } from './plant-card'; /** * Renders the Home Page view. * This component includes a hero section, a featured plants grid, * and a value proposition section. * * @param container - The HTMLElement where the home view will be rendered. */ export function renderHomeView(container: HTMLElement): void { // Clear existing content container.innerHTML = ''; // Create the main wrapper for the home view const homeWrapper = document.createElement('div'); homeWrapper.className = 'flex flex-col gap-16 pb-20 animate-in fade-in duration-700'; // 1. Hero Section const heroSection = document.createElement('section'); heroSection.className = 'relative w-full min-h-[70vh] flex items-center justify-center overflow-hidden rounded-3xl mt-4'; heroSection.innerHTML = `

Bring Nature Indoors

Discover a curated collection of premium indoor plants delivered straight to your doorstep. Transform your living space into a sanctuary today.

Shop Collection View Featured
`; // 2. Featured Section const featuredSection = document.createElement('section'); featuredSection.id = 'featured'; featuredSection.className = 'container-inner py-12'; const featuredHeader = document.createElement('div'); featuredHeader.className = 'flex flex-col md:flex-row md:items-end justify-between mb-12 gap-4'; featuredHeader.innerHTML = `

Featured Arrivals

Hand-picked favorites for your home office or living room.

Browse all plants `; const plantGrid = document.createElement('div'); plantGrid.className = 'grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-8'; // Generate 3 random plants for the featured section const featuredPlants: Plant[] = generatePlantCatalog(3); featuredPlants.forEach(plant => { const card = renderPlantCard(plant); plantGrid.appendChild(card); }); featuredSection.appendChild(featuredHeader); featuredSection.appendChild(plantGrid); // 3. Value Proposition Section const benefitsSection = document.createElement('section'); benefitsSection.className = 'bg-[var(--color-primary)] text-white py-20 rounded-[3rem] mx-4 md:mx-8'; benefitsSection.innerHTML = `

Guaranteed Healthy

Every plant is hand-picked and inspected by our botanists before shipping to ensure it arrives in peak condition.

Fast & Safe Delivery

Our specialized eco-friendly packaging keeps your plants secure and hydrated during their journey to your home.

Expert Care Support

New to plant parenting? Our experts are available 24/7 to help you with watering schedules and sunlight tips.

`; // Assemble the view homeWrapper.appendChild(heroSection); homeWrapper.appendChild(featuredSection); homeWrapper.appendChild(benefitsSection); container.appendChild(homeWrapper); }
