import { Product } from '../types'; import { formatCurrency, addToCart, getRandomRotationClass } from '../utils'; /** * Creates a DOM element representing a mystery plant product card. * Includes styling, data binding, and interactive "Add to Cart" functionality. * * @param product - The Product object containing details to display. * @returns An HTMLElement (article) ready to be appended to the DOM. */ export function createProductCard(product: Product): HTMLElement { const article = document.createElement('article'); // Apply base classes and a random rotation for the playful, organic vibe const rotationClass = getRandomRotationClass(); article.className = `bg-white rounded-2xl shadow-sm hover:shadow-xl transition-all duration-300 overflow-hidden flex flex-col h-full border border-gray-100 group ${rotationClass}`; // Generate the inner HTML structure article.innerHTML = `
${product.imageUrl ? `` : `` }
Mystery

${product.name}

${formatCurrency(product.price)}

${product.description}

Size: ${product.size || 'Standard'}
`; // Attach event listener for the Add to Cart button const addToCartBtn = article.querySelector('.add-to-cart-btn'); if (addToCartBtn) { addToCartBtn.addEventListener('click', (e) => { e.preventDefault(); e.stopPropagation(); // Add item to global cart state addToCart(product); // Provide visual feedback on the button const originalHTML = addToCartBtn.innerHTML; addToCartBtn.innerHTML = ` `; addToCartBtn.classList.add('bg-[var(--color-secondary)]'); addToCartBtn.classList.remove('bg-[var(--color-primary)]'); // Revert button state after 1 second setTimeout(() => { addToCartBtn.innerHTML = originalHTML; addToCartBtn.classList.remove('bg-[var(--color-secondary)]'); addToCartBtn.classList.add('bg-[var(--color-primary)]'); }, 1000); }); } return article; }
