import { CartItem } from '../types'; import { getCart, removeFromCart, updateCartQuantity, getCartTotal, formatCurrency, clearCart } from '../utils'; /** * Renders the Cart page view. * Displays all items in the user's shopping cart, allows quantity adjustments, * and provides a summary with a mock checkout process. * * @param container - The HTMLElement where the cart view will be rendered. */ export function renderCartView(container: HTMLElement): void { /** * Internal function to refresh the cart UI when data changes. */ const updateCartUI = () => { const cartItems = getCart(); const total = getCartTotal(); if (cartItems.length === 0) { renderEmptyState(container); } else { renderCartContent(container, cartItems, total); } }; // Initial render updateCartUI(); // Listen for global cart updates to keep this view in sync const handleUpdate = () => updateCartUI(); window.addEventListener('cart-updated', handleUpdate); // Cleanup listener if the container is cleared (basic memory management) const observer = new MutationObserver((mutations) => { mutations.forEach((mutation) => { mutation.removedNodes.forEach((node) => { if (node === container || container.contains(node)) { window.removeEventListener('cart-updated', handleUpdate); observer.disconnect(); } }); }); }); observer.observe(document.body, { childList: true, subtree: true }); } /** * Renders the UI for when the cart is empty. */ function renderEmptyState(container: HTMLElement): void { container.innerHTML = `

Your cart is empty

Looks like you haven't added any green friends to your collection yet. Explore our shop to find the perfect plant for your space.

Start Shopping
`; } /** * Renders the active cart content with items and summary. */ function renderCartContent(container: HTMLElement, items: CartItem[], total: number): void { container.innerHTML = `

Shopping Cart

${items.map(item => `

${item.plant.name}

${item.plant.sunlight} Light • ${item.plant.water} Water

${formatCurrency(item.plant.price)}

${item.quantity}
${formatCurrency(item.plant.price * item.quantity)}
`).join('')}
Continue Shopping

Order Summary

Subtotal ${formatCurrency(total)}
Shipping ${total > 50 ? 'FREE' : formatCurrency(9.99)}
Estimated Tax ${formatCurrency(total * 0.08)}
Total ${formatCurrency(total > 50 ? total * 1.08 : (total + 9.99) * 1.08)}
`; // Attach Event Listeners setupEventListeners(container, items); } /** * Sets up interaction logic for the cart view. */ function setupEventListeners(container: HTMLElement, items: CartItem[]): void { // Quantity Increase container.querySelectorAll('.plus-btn').forEach(btn => { btn.addEventListener('click', () => { const id = (btn as HTMLElement).dataset.id; const item = items.find(i => i.plant.id === id); if (item) updateCartQuantity(item.plant.id, item.quantity + 1); }); }); // Quantity Decrease container.querySelectorAll('.minus-btn').forEach(btn => { btn.addEventListener('click', () => { const id = (btn as HTMLElement).dataset.id; const item = items.find(i => i.plant.id === id); if (item && item.quantity > 1) { updateCartQuantity(item.plant.id, item.quantity - 1); } else if (item) { removeFromCart(item.plant.id); } }); }); // Remove Item container.querySelectorAll('.remove-btn').forEach(btn => { btn.addEventListener('click', () => { const id = (btn as HTMLElement).dataset.id; if (id) removeFromCart(id); }); }); // Clear Cart const clearBtn = container.querySelector('#clear-cart-btn'); clearBtn?.addEventListener('click', () => { if (confirm('Are you sure you want to empty your cart?')) { clearCart(); } }); // Mock Checkout const checkoutBtn = container.querySelector('#checkout-btn'); checkoutBtn?.addEventListener('click', () => { const btn = checkoutBtn as HTMLButtonElement; const originalText = btn.textContent; btn.disabled = true; btn.innerHTML = ` Processing... `; setTimeout(() => { alert('Thank you for your order! This is a demo store, so no payment was processed.'); clearCart(); // Redirect to home or success page window.history.pushState({}, '', '/'); window.dispatchEvent(new PopStateEvent('popstate')); }, 2000); }); }
