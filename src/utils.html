import { Product, CartItem, PotSize } from './types'; /** * Formats a number as a currency string (USD). */ export function formatCurrency(amount: number): string { return new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD', }).format(amount); } /** * Data pools for generating realistic random mystery products. */ const MYSTERY_NAMES = [ 'Mystery Seedling', 'Surprise Succulent', 'Unknown Aroid', 'Random Cutting', 'Enigmatic Fern', 'Secret Sprout', 'Hidden Gem Foliage', 'Blind Box Botanical' ]; const DESCRIPTIONS = [ 'A complete surprise! Planted in premium soil, just waiting to sprout and reveal its true identity.', 'We randomly selected a beautiful cutting from our greenhouse. What will it be? Only time will tell.', 'Perfect for the adventurous plant parent. Thrives on love, water, and a little bit of mystery.', 'A randomly chosen, fully rooted starter plant. Embrace the unknown and watch it grow!', 'Our botanists blindfolded themselves and picked this one just for you. Good luck!' ]; const POT_SIZES: PotSize[] = ['Small', 'Medium', 'Large', 'Extra Large']; const DIFFICULTIES: Product['difficultyHint'][] = ['Easy', 'Moderate', 'Expert', 'Complete Mystery']; /** * Generates a single random Product (Mystery Plant) object. */ export function generateRandomProduct(): Product { const id = Math.random().toString(36).substring(2, 9); const name = MYSTERY_NAMES[Math.floor(Math.random() * MYSTERY_NAMES.length)]; const potSize = POT_SIZES[Math.floor(Math.random() * POT_SIZES.length)]; // Base price depends roughly on pot size let basePrice = 15; if (potSize === 'Medium') basePrice = 25; if (potSize === 'Large') basePrice = 40; if (potSize === 'Extra Large') basePrice = 65; // Add a little random variance to the price const price = basePrice + Math.floor(Math.random() * 10); // Using Unsplash Source for high-quality placeholder images of wrapped/potted plants const imageUrl = `https://images.unsplash.com/photo-${1500000000000 + Math.floor(Math.random() * 1000000)}?auto=format&fit=crop&w=800&q=80&sig=${id}`; return { id, name: `${potSize} ${name}`, description: DESCRIPTIONS[Math.floor(Math.random() * DESCRIPTIONS.length)], price, imageUrl: `https://source.unsplash.com/featured/800x600?pottedplant,wrapped&sig=${id}`, potSize, difficultyHint: DIFFICULTIES[Math.floor(Math.random() * DIFFICULTIES.length)] }; } /** * Generates an array of random products. */ export function generateRandomProducts(count: number): Product[] { return Array.from({ length: count }, () => generateRandomProduct()); } /** * UI RANDOMIZER * Returns a random rotation utility class defined in style.css * to give elements a playfully "tossed" or organic look. */ export function getRandomRotationClass(): string { const classes = ['rotate-random-1', 'rotate-random-2', 'rotate-random-3', 'rotate-random-4']; return classes[Math.floor(Math.random() * classes.length)]; } /** * CART MANAGEMENT * Using localStorage for persistence and CustomEvents for reactivity. */ const CART_STORAGE_KEY = 'mystery_plant_cart'; /** * Dispatches a custom event so components can listen for cart changes. */ function notifyCartChange(): void { window.dispatchEvent(new CustomEvent('cart-updated')); } /** * Retrieves the current cart items from localStorage. */ export function getCartItems(): CartItem[] { const stored = localStorage.getItem(CART_STORAGE_KEY); if (!stored) return []; try { return JSON.parse(stored); } catch (e) { console.error('Failed to parse cart data', e); return []; } } /** * Saves the cart items to localStorage. */ export function saveCartItems(items: CartItem[]): void { localStorage.setItem(CART_STORAGE_KEY, JSON.stringify(items)); notifyCartChange(); } /** * Adds a product to the cart or increments its quantity. */ export function addToCart(product: Product): void { const items = getCartItems(); const existingItem = items.find(item => item.product.id === product.id); if (existingItem) { existingItem.quantity += 1; } else { items.push({ product, quantity: 1 }); } saveCartItems(items); } /** * Removes a product from the cart entirely. */ export function removeFromCart(productId: string): void { const items = getCartItems().filter(item => item.product.id !== productId); saveCartItems(items); } /** * Updates the quantity of a specific item in the cart. */ export function updateCartQuantity(productId: string, delta: number): void { const items = getCartItems(); const item = items.find(i => i.product.id === productId); if (item) { item.quantity += delta; if (item.quantity <= 0) { removeFromCart(productId); return; } saveCartItems(items); } } /** * Calculates the total number of items in the cart. */ export function getCartItemCount(): number { const items = getCartItems(); return items.reduce((sum, item) => sum + item.quantity, 0); } /** * Calculates the total price for the current cart. */ export function getCartTotal(): number { const items = getCartItems(); return items.reduce((sum, item) => sum + (item.product.price * item.quantity), 0); } /** * Clears all items from the cart. */ export function clearCart(): void { localStorage.removeItem(CART_STORAGE_KEY); notifyCartChange(); } /** * Helper to safely get an element from the DOM with type casting. */ export function getElement(selector: string): T | null { return document.querySelector(selector); }
