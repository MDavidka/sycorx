import { getCartItems, getCartTotal, formatCurrency, clearCart } from '../utils'; /** * Renders the Checkout page view into the provided container. * Displays the cart summary and a mock checkout form. * * @param container - The DOM element to mount the checkout view into. */ export function renderCheckoutView(container: HTMLElement): void { // Clear the container before rendering new content container.innerHTML = ''; const cartItems = getCartItems(); const wrapper = document.createElement('div'); wrapper.className = 'w-full flex flex-col animate-fade-in container-inner py-12 min-h-[70vh]'; // Handle Empty Cart State if (cartItems.length === 0) { wrapper.innerHTML = `

Your cart is empty

Looks like you haven't selected a mystery pot yet. Let's find the perfect vessel for your surprise plant.

Explore the Nursery
`; container.appendChild(wrapper); return; } // Calculate totals const subtotal = getCartTotal(); const shipping = subtotal > 50 ? 0 : 8.99; // Free shipping over $50 const total = subtotal + shipping; // Generate Cart Items HTML const cartItemsHTML = cartItems.map(item => `
${item.product.imageUrl ? `` : `` }

${item.product.name}

Qty: ${item.quantity} × ${formatCurrency(item.product.price)}

${formatCurrency(item.product.price * item.quantity)}
`).join(''); // Main Checkout Layout wrapper.innerHTML = `

Secure Checkout

Almost there! Just a few details before we plant your mystery.

1 Contact Information

2 Shipping Address

3 Payment Details

This is a mock checkout. No real payment will be processed.

By placing your order, you agree to our Terms of Service and acknowledge that the plant you receive is a complete mystery.

Order Summary

${cartItemsHTML}
Subtotal ${formatCurrency(subtotal)}
Shipping ${shipping === 0 ? 'Free' : formatCurrency(shipping)}
${shipping > 0 ? `
Add ${formatCurrency(50 - subtotal)} more for free shipping!
` : ''}
Total ${formatCurrency(total)}

Healthy Sprout Guarantee

If your mystery seed doesn't sprout within 30 days, we'll send you a new one for free.

`; container.appendChild(wrapper); // Handle Form Submission const form = wrapper.querySelector('#checkout-form') as HTMLFormElement; if (form) { form.addEventListener('submit', (e) => { e.preventDefault(); // Simulate processing time const submitBtn = form.querySelector('button[type="submit"]') as HTMLButtonElement; const originalBtnText = submitBtn.innerHTML; submitBtn.disabled = true; submitBtn.innerHTML = ` Processing... `; submitBtn.classList.add('opacity-75', 'cursor-not-allowed'); setTimeout(() => { // Clear the cart clearCart(); // Dispatch custom event to update header cart count window.dispatchEvent(new Event('cart-updated')); // Render Success State container.innerHTML = `

Order Confirmed!

Your mystery pot is being prepared. We'll email you the tracking details once it ships. Get ready for a surprise!

Order Number

#MYST-${Math.floor(10000 + Math.random() * 90000)}

Return Home
`; // Scroll to top window.scrollTo({ top: 0, behavior: 'smooth' }); }, 1500); }); } }
