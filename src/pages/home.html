import { generateRandomProducts } from '../utils'; import { createProductCard } from '../components/productCard'; /** * Renders the Home page view into the provided container. * Includes a hero section, a "How it Works" section, and a featured products grid. * * @param container - The DOM element to mount the home view into. */ export function renderHomeView(container: HTMLElement): void { // Clear the container before rendering new content container.innerHTML = ''; // Create a wrapper for the home page content const wrapper = document.createElement('div'); wrapper.className = 'w-full flex flex-col animate-fade-in'; // 1. Hero Section const heroHTML = `
Botanical Roulette

Embrace the Surprise.

You pick the pot. We plant the seed. What grows is a complete mystery until it sprouts. Add a little unpredictable nature to your home.

Shop Mystery Pots
`; // 2. How It Works Section const howItWorksHTML = `

How It Works

1. Choose a Pot

Select from our curated collection of premium terracotta, ceramic, and eco-resin pots.

2. We Plant a Mystery

Our botanists select a random, healthy seed or cutting and plant it in your chosen vessel.

3. Watch it Grow

Care for your new friend and discover what beautiful plant unfolds over time.

`; // 3. Featured Products Section const featuredHTML = `

Featured Mysteries

Our most popular unknown botanical wonders.

View all →
View all →
`; // Combine all sections and append to the wrapper wrapper.innerHTML = heroHTML + howItWorksHTML + featuredHTML; container.appendChild(wrapper); // Dynamically generate and inject 3 featured products const featuredGrid = wrapper.querySelector('#featured-grid'); if (featuredGrid) { const featuredProducts = generateRandomProducts(3); featuredProducts.forEach(product => { const card = createProductCard(product); featuredGrid.appendChild(card); }); } }
