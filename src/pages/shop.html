import { generateRandomProducts } from '../utils'; import { createProductCard } from '../components/productCard'; /** * Renders the Shop page view into the provided container. * Displays a full grid of available mystery plant pots. * * @param container - The DOM element to mount the shop view into. */ export function renderShopView(container: HTMLElement): void { // Clear the container before rendering new content container.innerHTML = ''; // Create a wrapper for the shop page content const wrapper = document.createElement('div'); wrapper.className = 'w-full flex flex-col animate-fade-in container-inner py-12 min-h-screen'; // Shop Header Section const headerHTML = `
Complete Collection

The Mystery Nursery

Browse our collection of premium vessels. Each one comes pre-planted with a randomly selected, healthy botanical surprise. Choose your pot size and style, and let nature do the rest.

`; // Grid Container const gridHTML = `
`; // Combine header and grid wrapper.innerHTML = headerHTML + gridHTML; container.appendChild(wrapper); // Dynamically generate and inject 12 products for the shop catalog const grid = wrapper.querySelector('#shop-grid'); if (grid) { // Generate a larger batch of products for the main shop page const products = generateRandomProducts(12); products.forEach((product, index) => { const card = createProductCard(product); // Stagger the entrance animation slightly for a cascading effect card.style.opacity = '0'; card.style.animation = `fadeInUp 0.5s ease forwards ${index * 0.05}s`; grid.appendChild(card); }); } // Add the keyframes for the staggered animation dynamically if not present if (!document.getElementById('shop-animations')) { const style = document.createElement('style'); style.id = 'shop-animations'; style.textContent = ` @keyframes fadeInUp { from { opacity: 0; transform: translateY(20px); } to { opacity: 1; transform: translateY(0); } } `; document.head.appendChild(style); } }
